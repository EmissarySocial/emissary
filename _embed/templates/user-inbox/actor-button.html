{{- $url := .QueryParam "url" -}} {{- $following := .AmFollowing $url -}} {{- if not $following.IsNew -}} {{- $subRenderer := .SubRenderer $following -}} {{- template "actor-button-follow" $subRenderer -}} {{- else -}} {{- $rule := .HasRule "ACTOR" $url -}} {{- if $rule.IsNew -}} {{- template "actor-button-none" . -}} {{- else if eq "MUTE" $rule.Action -}} {{- $subRenderer := .SubRenderer $rule -}} {{- template "actor-button-mute" $subRenderer -}} {{- else if eq "BLOCK" $rule.Action -}} {{- $subRenderer := .SubRenderer $rule -}} {{- template "actor-button-block" $subRenderer -}} {{- else -}} {{- template "actor-button-none" . -}} {{- end -}} {{- end -}}
